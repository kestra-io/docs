FileTransform 

type: "io.kestra.plugin.scripts.groovy.FileTransform"

# Examples

id: "file_transform"
type: "io.kestra.plugin.scripts.groovy.FileTransform"
from: "{{ outputs.avro-to-gcs }}"
script: |
  logger.info('row: {}', row)

  if (row.get('name') == 'richard') {
    row = null
  } else {
    row.put('email', row.get('name') + '@kestra.io')
  }
